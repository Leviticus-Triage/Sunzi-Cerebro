```javascript
  'viewer': [

    '*' // Full System Access
  ]
};

// Granular Permission Enforcement
const checkEnterprisePermission = (requiredPermission, userRole, tenantId) => {
  // Multi-Tenant Permission Validation
  const tenantPermissions = getTenantPermissions(tenantId);
  const rolePermissions = ENTERPRISE_PERMISSIONS[userRole] || [];

  return hasPermission(rolePermissions, requiredPermission) &&
         tenantPermissions.includes(requiredPermission);
};
```
